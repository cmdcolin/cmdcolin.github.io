42 /*
43  * Slow but simple huffman decoder to start with.
44  * Read a bit at a time, keeping track of {length, value}
45  * eg. 1 1 0 1 => {1,1}, {2,3}, {3,6}, {4,13}
46  *
47  * Keep track of this through the huffman code table.
48  * For fast scanning we have an index of where the first code of length X
49  * appears.
50  */

154 /*
155  * Check that enough bits are left in a block to satisy a bit-based decoder.
156  * Return 0 if there are enough
157  * 1 if not.
158  */
159 
160 static inline int cram_not_enough_bits(cram_block *blk, int nbits) {
161  if (nbits < 0 ||
162  blk->byte >= blk->uncomp_size ||
163  (blk->uncomp_size - blk->byte <= INT32_MAX / 8 + 1 &&
164  (blk->uncomp_size - blk->byte) * 8 + blk->bit - 7 < nbits)) {
165  return 1;
166  }
167  return 0;
168 }
