25 /*
26  Regions indexing with an optional payload. Inspired by samtools/bedidx.c.
27  This code is intended as future replacement of bcf_sr_regions_t.
28 
29  Example of usage:
30 
31  // Init the parser and print regions. In this example the payload is a
32  // pointer to a string. For the description of parse_custom and
33  // free_custom functions, see regidx_parse_f and regidx_free_f below,
34  // and for working example see test/test-regidx.c.
35  regidx_t *idx = regidx_init(in_fname,parse_custom,free_custom,sizeof(char*),NULL);
36 
37  // Query overlap with chr:from-to
38  regitr_t itr;
39  if ( regidx_overlap(idx, chr,from,to, &itr) ) printf("There is an overlap!\n");
40 
41  while ( REGITR_OVERLAP(itr,from,to) )
42  {
43  printf("[%d,%d] overlaps with [%d,%d], payload=%s\n", from,to,
44  REGITR_START(itr), REGITR_END(itr), REGITR_PAYLOAD(itr,char*));
45  itr.i++;
46  }
47 
48  regidx_destroy(regs);
49 */

80 /*
81  * regidx_parse_f - Function to parse one input line, such as regidx_parse_bed
82  * or regidx_parse_tab below. The function is expected to set `chr_from` and
83  * `chr_to` to point to first and last character of chromosome name and set
84  * coordinates `reg->start` and `reg->end` (0-based, inclusive). If
85  * regidx_init() was called with non-zero payload_size, the `payload` points
86  * to a memory location of the payload_size and `usr` is data passed to
87  * regidx_init(). Any memory allocated by the function will be freed by
88  * regidx_free_f on regidx_destroy().
89  *
90  * Return value: 0 on success, -1 to skip a record, -2 on fatal error.
91  */
92 typedef int (*regidx_parse_f)(const char *line, char **chr_beg, char **chr_end, reg_t *reg, void *payload, void *usr);
93 typedef void (*regidx_free_f)(void *payload);
