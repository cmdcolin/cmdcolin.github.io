67 /* === Dictionary ===
68 
69  The header keeps three dictonaries. The first keeps IDs in the
70  "FILTER/INFO/FORMAT" lines, the second keeps the sequence names and lengths
71  in the "contig" lines and the last keeps the sample names. bcf_hdr_t::dict[]
72  is the actual hash table, which is opaque to the end users. In the hash
73  table, the key is the ID or sample name as a C string and the value is a
74  bcf_idinfo_t struct. bcf_hdr_t::id[] points to key-value pairs in the hash
75  table in the order that they appear in the VCF header. bcf_hdr_t::n[] is the
76  size of the hash table or, equivalently, the length of the id[] arrays.
77 */

187 /*
188  The bcf1_t structure corresponds to one VCF/BCF line. Reading from VCF file
189  is slower because the string is first to be parsed, packed into BCF line
190  (done in vcf_parse), then unpacked into internal bcf1_t structure. If it
191  is known in advance that some of the fields will not be required (notably
192  the sample columns), parsing of these can be skipped by setting max_unpack
193  appropriately.
194  Similarly, it is fast to output a BCF line because the columns (kept in
195  shared.s, indiv.s, etc.) are written directly by bcf_write, whereas a VCF
196  line must be formatted in vcf_format.
197  */
198 typedef struct {
199  int32_t rid; // CHROM
200  int32_t pos; // POS
201  int32_t rlen; // length of REF
202  float qual; // QUAL
203  uint32_t n_info:16, n_allele:16;
204  uint32_t n_fmt:8, n_sample:24;
205  kstring_t shared, indiv;
206  bcf_dec_t d; // lazy evaluation: $d is not generated by bcf_read(), but by explicitly calling bcf_unpack()
207  int max_unpack; // Set to BCF_UN_STR, BCF_UN_FLT, or BCF_UN_INFO to boost performance of vcf_parse when some of the fields won't be needed
208  int unpacked; // remember what has been unpacked to allow calling bcf_unpack() repeatedly without redoing the work
209  int unpack_size[3]; // the original block size of ID, REF+ALT and FILTER
210  int errcode; // one of BCF_ERR_* codes
211 } bcf1_t;

740 /*
741  Note that in contrast with BCFv2.1 specification, HTSlib implementation
742  allows missing values in vectors. For integer types, the values 0x80,
743  0x8000, 0x80000000 are interpreted as missing values and 0x81, 0x8001,
744  0x80000001 as end-of-vector indicators. Similarly for floats, the value of
745  0x7F800001 is interpreted as a missing value and 0x7F800002 as an
746  end-of-vector indicator.
747  Note that the end-of-vector byte is not part of the vector.
748 
749  This trial BCF version (v2.2) is compatible with the VCF specification and
750  enables to handle correctly vectors with different ploidy in presence of
751  missing values.
752  */
753 #define bcf_int8_vector_end (INT8_MIN+1)
754 #define bcf_int16_vector_end (INT16_MIN+1)
755 #define bcf_int32_vector_end (INT32_MIN+1)
756 #define bcf_str_vector_end 0
757 #define bcf_int8_missing INT8_MIN
758 #define bcf_int16_missing INT16_MIN
759 #define bcf_int32_missing INT32_MIN
760 #define bcf_str_missing 0x07

784 static inline void bcf_format_gt(bcf_fmt_t *fmt, int isample, kstring_t *str)
785 {
786  #define BRANCH(type_t, missing, vector_end) { \
787  type_t *ptr = (type_t*) (fmt->p + isample*fmt->size); \
788  int i; \
789  for (i=0; i<fmt->n && ptr[i]!=vector_end; i++) \
790  { \
791  if ( i ) kputc("/|"[ptr[i]&1], str); \
792  if ( !(ptr[i]>>1) ) kputc('.', str); \
793  else kputw((ptr[i]>>1) - 1, str); \
794  } \
795  if (i == 0) kputc('.', str); \
796  }
797  switch (fmt->type) {
798  case BCF_BT_INT8: BRANCH(int8_t, bcf_int8_missing, bcf_int8_vector_end); break;
799  case BCF_BT_INT16: BRANCH(int16_t, bcf_int16_missing, bcf_int16_vector_end); break;
800  case BCF_BT_INT32: BRANCH(int32_t, bcf_int32_missing, bcf_int32_vector_end); break;
801  default: fprintf(stderr,"FIXME: type %d in bcf_format_gt?\n", fmt->type); abort(); break;
802  }
803  #undef BRANCH
804 }

806 static inline void bcf_enc_size(kstring_t *s, int size, int type)
807 {
808  if (size >= 15) {
809  kputc(15<<4|type, s);
810  if (size >= 128) {
811  if (size >= 32768) {

823  }

825 }
