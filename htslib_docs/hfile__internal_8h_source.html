34 struct hFILE_backend {
35  /* As per read(2), returning the number of bytes read (possibly 0) or
36  negative (and setting errno) on errors. Front-end code will call this
37  repeatedly if necessary to attempt to get the desired byte count. */
38  ssize_t (*read)(hFILE *fp, void *buffer, size_t nbytes) HTS_RESULT_USED;
39 
40  /* As per write(2), returning the number of bytes written or negative (and
41  setting errno) on errors. Front-end code will call this repeatedly if
42  necessary until the desired block is written or an error occurs. */
43  ssize_t (*write)(hFILE *fp, const void *buffer, size_t nbytes)
44  HTS_RESULT_USED;
45 
46  /* As per lseek(2), returning the resulting offset within the stream or
47  negative (and setting errno) on errors. */
48  off_t (*seek)(hFILE *fp, off_t offset, int whence) HTS_RESULT_USED;
49 
50  /* Performs low-level flushing, if any, e.g., fsync(2); for writing streams
51  only. Returns 0 for success or negative (and sets errno) on errors. */
52  int (*flush)(hFILE *fp) HTS_RESULT_USED;
53 
54  /* Closes the underlying stream (for output streams, the buffer will
55  already have been flushed), returning 0 for success or negative (and
56  setting errno) on errors, as per close(2). */
57  int (*close)(hFILE *fp) HTS_RESULT_USED;
58 };
