110 static inline int safe_itf8_get(const char *cp, const char *endp,
111  int32_t *val_p) {
112  const unsigned char *up = (unsigned char *)cp;
113 
114  if (endp - cp < 5 &&
115  (cp >= endp || endp - cp < itf8_bytes[up[0]>>4])) return 0;
116 
117  if (up[0] < 0x80) {
118  *val_p = up[0];
119  return 1;
120  } else if (up[0] < 0xc0) {
121  *val_p = ((up[0] <<8) | up[1]) & 0x3fff;
122  return 2;
123  } else if (up[0] < 0xe0) {
124  *val_p = ((up[0]<<16) | (up[1]<< 8) | up[2]) & 0x1fffff;
125  return 3;
126  } else if (up[0] < 0xf0) {
127  *val_p = ((up[0]<<24) | (up[1]<<16) | (up[2]<<8) | up[3]) & 0x0fffffff;
128  return 4;
129  } else {
130  *val_p = ((up[0] & 0x0f)<<28) | (up[1]<<20) | (up[2]<<12) | (up[3]<<4) | (up[4] & 0x0f);
131  return 5;
132  }
133 }

247 /* Request block to be at least 'l' bytes long */
248 #define BLOCK_RESIZE(b,l) \
249  do { \

256 /* Ensure the block can hold at least another 'l' bytes */
257 #define BLOCK_GROW(b,l) BLOCK_RESIZE((b), BLOCK_SIZE((b)) + (l))

259 /* Append string 's' of length 'l' */

283 static inline unsigned char *append_uint32(unsigned char *cp, uint32_t i) {
284  uint32_t j;
285 
286  if (i == 0) {
287  *cp++ = '0';
288  return cp;
289  }
290 
291  if (i < 100) goto b1;
292  if (i < 10000) goto b3;
293  if (i < 1000000) goto b5;
294  if (i < 100000000) goto b7;
295 
296  if ((j = i / 1000000000)) {*cp++ = j + '0'; i -= j*1000000000; goto x8;}
297  if ((j = i / 100000000)) {*cp++ = j + '0'; i -= j*100000000; goto x7;}
298  b7:if ((j = i / 10000000)) {*cp++ = j + '0'; i -= j*10000000; goto x6;}
299  if ((j = i / 1000000)) {*cp++ = j + '0', i -= j*1000000; goto x5;}
300  b5:if ((j = i / 100000)) {*cp++ = j + '0', i -= j*100000; goto x4;}
301  if ((j = i / 10000)) {*cp++ = j + '0', i -= j*10000; goto x3;}
302  b3:if ((j = i / 1000)) {*cp++ = j + '0', i -= j*1000; goto x2;}
303  if ((j = i / 100)) {*cp++ = j + '0', i -= j*100; goto x1;}
304  b1:if ((j = i / 10)) {*cp++ = j + '0', i -= j*10; goto x0;}
305  if (i) *cp++ = i + '0';
306  return cp;
307 
308  x8: *cp++ = i / 100000000 + '0', i %= 100000000;
309  x7: *cp++ = i / 10000000 + '0', i %= 10000000;
310  x6: *cp++ = i / 1000000 + '0', i %= 1000000;
311  x5: *cp++ = i / 100000 + '0', i %= 100000;
312  x4: *cp++ = i / 10000 + '0', i %= 10000;
313  x3: *cp++ = i / 1000 + '0', i %= 1000;
314  x2: *cp++ = i / 100 + '0', i %= 100;
315  x1: *cp++ = i / 10 + '0', i %= 10;
316  x0: *cp++ = i + '0';

335 static inline unsigned char *append_uint64(unsigned char *cp, uint64_t i) {
336  uint64_t j;
337 
338  if (i <= 0xffffffff)
339  return append_uint32(cp, i);
340 
341  if ((j = i/1000000000) > 1000000000) {
342  cp = append_uint32(cp, j/1000000000);
343  j %= 1000000000;
344  cp = append_sub32(cp, j);
345  } else {
346  cp = append_uint32(cp, i / 1000000000);
347  }
348  cp = append_sub32(cp, i % 1000000000);
349 
350  return cp;
