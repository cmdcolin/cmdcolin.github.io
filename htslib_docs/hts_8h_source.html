453 static inline int hts_reg2bin(int64_t beg, int64_t end, int min_shift, int n_lvls)
454 {
455  int l, s = min_shift, t = ((1<<((n_lvls<<1) + n_lvls)) - 1) / 7;
456  for (--end, l = n_lvls; l > 0; --l, s += 3, t -= 1<<((l<<1)+l))
457  if (beg>>s == end>>s) return t + (beg>>s);
458  return 0;
459 }

461 static inline int hts_bin_bot(int bin, int n_lvls)
462 {
463  int l, b;
464  for (l = 0, b = bin; b; ++l, b = hts_bin_parent(b)); // compute the level of bin
465  return (bin - hts_bin_first(l)) << (n_lvls - l) * 3;
466 }
